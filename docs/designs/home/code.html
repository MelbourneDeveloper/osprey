A memory-safe, high-concurrency language designed for the modern multicore era. Osprey blends the control of C++ with the safety of Rust and the ergonomics of TypeScript.

import std.io;
import std.concurrency;

pub func main() {
    // Talon concurrency in action
    spawn {
        io.println("Hello from a light-weight talon!");
    }
